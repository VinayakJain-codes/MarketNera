<script>
        // Micro-interaction: Update Star Fill on Rating Hover (Simulated)
        const stars = document.querySelectorAll('.material-symbols-outlined[style*="star"]');
        stars.forEach((star, idx) => {
            star.addEventListener('click', () => {
                stars.forEach((s, i) => {
                    s.style.fontVariationSettings = i <= idx ? "'FILL' 1" : "'FILL' 0";
                });
            });
        });

        // Floating Header Effect
        window.addEventListener('scroll', () => {
            const header = document.querySelector('header');
            if (window.scrollY > 20) {
                header.classList.add('shadow-md');
                header.classList.add('bg-white/80');
                header.classList.add('backdrop-blur-md');
            } else {
                header.classList.remove('shadow-md');
                header.classList.remove('bg-white/80');
                header.classList.remove('backdrop-blur-md');
            }
        });
    </script>
